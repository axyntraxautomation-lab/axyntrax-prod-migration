Stack: PostgreSQL + Supabase free tier + Prisma ORM

TABLAS BASE (comunes a todos los rubros):
```sql
-- Empresa/Negocio
CREATE TABLE empresa (
  id UUID PRIMARY KEY,
  nombre TEXT,
  rubro TEXT,
  logo_url TEXT,
  config JSONB,   -- colores, dominio, credenciales Yape/Plin
  created_at TIMESTAMP
);

-- Clientes del negocio
CREATE TABLE clientes (
  id UUID PRIMARY KEY,
  nombre TEXT,
  telefono TEXT UNIQUE,
  email TEXT,
  rubro_data JSONB,  -- datos específicos del rubro
  historial JSONB,
  created_at TIMESTAMP
);

-- Servicios/Productos (adaptable)
CREATE TABLE catalogo (
  id UUID PRIMARY KEY,
  empresa_id UUID REFERENCES empresa,
  nombre TEXT,
  tipo TEXT,       -- "servicio" | "producto"
  precio DECIMAL,
  duracion INT,    -- minutos (para servicios)
  stock INT,       -- unidades (para productos)
  activo BOOLEAN DEFAULT true
);

-- Transacciones unificadas
CREATE TABLE transacciones (
  id UUID PRIMARY KEY,
  empresa_id UUID REFERENCES empresa,
  cliente_id UUID REFERENCES clientes,
  tipo TEXT,       -- "venta" | "gasto" | "pago"
  monto DECIMAL,
  metodo_pago TEXT,-- "yape"|"plin"|"efectivo"|"tarjeta"
  estado TEXT,     -- "pendiente"|"confirmado"|"cancelado"
  metadata JSONB,  -- datos específicos del rubro
  created_at TIMESTAMP
);

-- Agenda/Citas/Órdenes
CREATE TABLE agenda (
  id UUID PRIMARY KEY,
  empresa_id UUID REFERENCES empresa,
  cliente_id UUID REFERENCES clientes,
  servicio_id UUID REFERENCES catalogo,
  fecha TIMESTAMP,
  estado TEXT,
  notas TEXT,
  recordatorio_enviado BOOLEAN DEFAULT false
);

-- Stock/Inventario
CREATE TABLE inventario (
  id UUID PRIMARY KEY,
  empresa_id UUID REFERENCES empresa,
  producto TEXT,
  cantidad DECIMAL,
  unidad TEXT,
  minimo_alerta DECIMAL,
  updated_at TIMESTAMP
);
```

Backup automático a Google Drive cada 24h.
